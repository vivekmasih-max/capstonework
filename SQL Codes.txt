1)SELECT * FROM BUDDYFLYDATA
WHERE `ARRIVAL TIME` BETWEEN '2024-09-01 09:30:55' AND '2024-11-30 04:40:30';


2)SELECT * FROM BUDDYFLYDATA LIMIT 10 OFFSET 10;

3)SELECT `Types_of_AirCraft`, COUNT(`YEARS OF RUNNING`) AS PERFORMANCE_YEARS
FROM BUDDYFLYDATA
GROUP BY `Types_of_AirCraft`;


4)SELECT 
    DISTINCT(BUDDYFLYDATA.MANUFACTURER),
    SUM(CASE WHEN CATEGORYDATA.CATEGORY = 'DOMESTIC' THEN 1 ELSE 0 END) AS DOMESTIC_VEHICLE,
    SUM(CASE WHEN CATEGORYDATA.CATEGORY = 'INTERNATIONAL' THEN 1 ELSE 0 END) AS INTERNATIONAL_VEHICLE
FROM CATEGORYDATA
INNER JOIN BUDDYFLYDATA
    ON CATEGORYDATA.IDNEW = BUDDYFLYDATA.ID
GROUP BY BUDDYFLYDATA.MANUFACTURER;


5)SELECT SUM(`PassengerCount`) AS Total_Passengers
FROM BUDDYFLYDATA
WHERE `Departure Date` > (
    SELECT MAX(`Departure Date`)
    FROM BUDDYFLYDATA
    WHERE `Flight Status` = 'Delayed'
);
